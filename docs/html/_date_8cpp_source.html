38// Description :

40// Provides a Date class which represents dates as
41// Julian day numbers ( days since 1 Jan 4713 BC ).
42// This class can handle all dates from 1 Jan 4713BC to 31 Dec 4999AD.
43//
44// Note: Years AD are positive, years BC are negative. There is
45// no year 0AD, it goes from 1BC to 1AD. A year of 0 will be treated
46// as 1BC. No attempt is made to validate ranges. Physical errors
47// will not result from insane day-month combinations like the 87th
48// day of the 45th month, but the results will obviously not make
49// much sense.

77// Internal routine that does the physical conversion
78// from YMD form to Julian day number.

80long Date::YmdToJd( const int iYear, const int iMonth, const int iDay )
81{
82 long jul_day;
83
84#ifndef JULDATE_USE_ALTERNATE_METHOD
85
86 int a,b;
87 int year = iYear, month = iMonth, day = iDay;
88 float year_corr;
89
90 if ( year < 0 )
91 year++;
92 year_corr = ( year > 0 ? 0.0 : 0.75 );
93 if ( month <= 2 )
94 {
95 year--;
96 month += 12;
97 }
98 b = 0;
99 if ( year * 10000.0 + month * 100.0 + day >= 15821015.0 )
100 {
101 a = year / 100;
102 b = 2 - a + a / 4;
103 }
104 jul_day = (long) ( 365.25 * year - year_corr ) +
105 (long) ( 30.6001 * ( month + 1 ) ) + day + 1720995L + b;
106
107#else
108
109 long lmonth = (long) iMonth, lday = (long) iDay, lyear = (long) iYear;
110
111 // Adjust BC years
112 if ( lyear < 0 )
113 lyear++;
114
115 jul_day = lday - 32075L +
116 1461L * ( lyear + 4800L + ( lmonth - 14L ) / 12L ) / 4L +
117 367L * ( lmonth - 2L - ( lmonth - 14L ) / 12L * 12L ) / 12L -
118 3L * ( ( lyear + 4900L + ( lmonth - 14L ) / 12L ) / 100L ) / 4L;
119
120#endif
121
122 return jul_day;
123}

126// Function : JdToYmd
127//
128// Author : Todd Knarr
129//
130// Creation date : 29 Nov 1995
131//
132// Parameters : long julian day, pointers to int year, month, day
133//
134// Return values : none
135//
136// Description :
137//
138// Internal routine that reverses the conversion, turning a Julian
139// day number into YMD values.
140//
141void Date::JdToYmd( const long lJD, int *piYear, int *piMonth, int *piDay )
142{
143#ifndef JULDATE_USE_ALTERNATE_METHOD
144
145 long a, b, c, d, e, z, alpha;
146
147 z = lJD;
148 if ( z < 2299161L )
149 a = z;
150 else
151 {
152 alpha = (long) ( ( z - 1867216.25 ) / 36524.25 );
153 a = z + 1 + alpha - alpha / 4;
154 }
155 b = a + 1524;
156 c = (long) ( ( b - 122.1 ) / 365.25 );
157 d = (long) ( 365.25 * c );
158 e = (long) ( ( b - d ) / 30.6001 );
159 *piDay = (int) b - d - (long) ( 30.6001 * e );
160 *piMonth = (int) ( e < 13.5 ) ? e - 1 : e - 13;
161 *piYear = (int) ( *piMonth > 2.5 ) ? ( c - 4716 ) : c - 4715;
162 if ( *piYear <= 0 )
163 *piYear -= 1;
164
165#else
166
167 long t1, t2, yr, mo;
168
169 t1 = lJD + 68569L;
170 t2 = 4L * t1 / 146097L;
171 t1 = t1 - ( 146097L * t2 + 3L ) / 4L;
172 yr = 4000L * ( t1 + 1L ) / 1461001L;
173 t1 = t1 - 1461L * yr / 4L + 31L;
174 mo = 80L * t1 / 2447L;
175 *piDay = (int) ( t1 - 2447L * mo / 80L );
176 t1 = mo / 11L;
177 *piMonth = (int) ( mo + 2L - 12L * t1 );
178 *piYear = (int) ( 100L * ( t2 - 49L ) + yr + t1 );
179
180 // Correct for BC years
181 if ( *piYear <= 0 )
182 *piYear -= 1;
183
184#endif
185
186 return;
187}
